                                            Updating the listing of supported devices

                                            Copy and paste the string generated by the this script into this help file:

                                            (
MKtlDesc.loadDescs;

MKtlDesc.deviceTypesUsed.sort.collect { |type|
    var str = "subsection:: %\nlist::\n".format(type);
    MKtlDesc.allDescs.select { |d| d.fullDesc.deviceType == type }
    .do { |dev| str = str ++ "## %\n".format(dev.cs); };
    str = str ++ "::\n\n";
}.join;
)
