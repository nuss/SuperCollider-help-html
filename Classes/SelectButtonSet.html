        Description

        Radio button style set, fashioned as a single object.

        (

    SelectButtonSet( nil,  [ "one", "two","three","four"] ,
        { arg selectedIndex,selectButtonSet;
            [ selectedIndex, selectButtonSet ].postln;
        },width: 60
    )
)

(
    SelectButtonSet( nil,
        16 , //  make 16 buttons
        { arg selectedIndex,selectButtonSet;
            [ selectedIndex, selectButtonSet ].postln;
        }
    )
)
