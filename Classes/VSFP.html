      Description

      Varispeed sound file player. This is for playing long soundfiles from disk. This is not a sample player.

      (
v = VSFP(path/to/file);
v.play;
v.pchRatio = 0.4;
v.pchRatio = 1.9;
)


(

VSFP.getNew({ arg v;
    Sheet({ arg layout;
        v.topGui(layout);
        NumberEditor(1.0,[0.01,32.0])
            .action_({ arg value; v.pchRatio = value; })
            .gui(layout)
    })
})

)
